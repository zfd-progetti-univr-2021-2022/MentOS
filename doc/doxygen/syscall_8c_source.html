113 void syscall_handler(pt_regs *f)
114 {
115  // Saves current interrupt stack frame
116  current_interrupt_stack_frame = f;
117 
118  // dbg_print_regs(f);
119  // Save current process fpu state.
120  switch_fpu();
121 
122  // The index of the requested system call.
123  uint32_t sc_index = f->eax;
124 
125  // The result of the system call.
126  int ret;
127  if (sc_index >= SYSCALL_NUMBER) {
128  ret = ENOSYS;
129  } else {
130  uintptr_t ptr = (uintptr_t)sys_call_table[sc_index];
131 
132  SystemCall func = (SystemCall)ptr;
133 
134  uint32_t arg0 = f->ebx;
135  uint32_t arg1 = f->ecx;
136  uint32_t arg2 = f->edx;
137  uint32_t arg3 = f->esi;
138  uint32_t arg4 = f->edi;
139  if ((sc_index == __NR_fork) ||
140  (sc_index == __NR_clone) ||
141  (sc_index == __NR_execve) ||
142  (sc_index == __NR_sigreturn)) {
143  arg0 = (uintptr_t)f;
144  }
145  ret = func(arg0, arg1, arg2, arg3, arg4);
146  }
147  f->eax = ret;
148 
149  // Schedule next process.
150  scheduler_run(f);
151  // Restore fpu state.
152  unswitch_fpu();
153 }
