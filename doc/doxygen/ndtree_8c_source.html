193 static ndtree_node_t *__ndtree_tree_find_rec(ndtree_t *tree, ndtree_tree_cmp_f cmp, void *value, ndtree_node_t *node)
194 {
195  ndtree_node_t *result = NULL;
196  if (tree && cmp && node && value) {
197  if (cmp(tree, node->value, value) == 0) {
198  result = node;
199  } else if (!list_head_empty(&node->children)) {
200  list_for_each_decl(it, &node->children)
201  {
202  ndtree_node_t *child = list_entry(it, ndtree_node_t, siblings);
203  if ((result = __ndtree_tree_find_rec(tree, cmp, value, child)) != NULL) {
204  break;
205  }
206  }
207  }
208  }
209  return result;

219 ndtree_node_t *ndtree_node_find(ndtree_t *tree, ndtree_node_t *node, ndtree_tree_cmp_f cmp, void *value)
220 {
221  if (tree && node && value) {
222  // Check only if the node has children.
223  if (!list_head_empty(&node->children)) {
224  // Check which compare function we need to use.
225  ndtree_tree_cmp_f cmp_fun = cmp ? cmp : tree->cmp;
226  // If neither the tree nor the function argument are valid, rollback to the
227  // default comparison function.
228  if (cmp_fun == NULL)
229  cmp_fun = __ndtree_tree_node_cmp_ptr_cb;
230  // Iterate throught the children.
231  list_for_each_decl(it, &node->children)
232  {
233  ndtree_node_t *child = list_entry(it, ndtree_node_t, siblings);
234  if (cmp_fun(tree, child->value, value) == 0)
235  return child;
236  }
237  }
238  }
239  return NULL;
240 }

249 int ndtree_tree_remove_node_with_cb(ndtree_t *tree, ndtree_node_t *node, ndtree_tree_node_f node_cb)
250 {
251  if (tree && node) {
252  // Remove the node from the parent list.
253  list_head_del(&node->siblings);
254  // If the node has children, we need to migrate them.
255  if (!list_head_empty(&node->children)) {
256  // The new parent, by default it is NULL.
257  ndtree_node_t *new_parent = NULL;
258  // The new list, by default it is the list of orphans of the tree.
259  list_head *new_list = &tree->orphans;
260  // If the found node has a parent, we need to set the variables
261  // so that we can migrate the children.
262  if (node->parent) {
263  new_parent = node->parent;
264  new_list = &node->parent->children;
265  }
266  // Migrate the children.
267  list_for_each_decl(it, &node->children)
268  {
269  ndtree_node_t *child = list_entry(it, ndtree_node_t, siblings);
270  child->parent = new_parent;
271  }
272  // Merge the lists.
273  list_head_merge(new_list, &node->children);
274  }
275  if (node_cb)
276  node_cb(tree, node);
277  else
278  ndtree_node_dealloc(node);
279  --tree->size;
280  return 1;
281  }
282  return 0;
283 }
