152 static void __update_task_statistics(task_struct *task)
153 {
154  assert(task && "Current task is not valid.");
155 
156  // While periodic task is under analysis is executed with aperiodic
157  // scheduler and can be preempted by a "true" periodic task.
158  // We need to sum all the execution spots to calculate the WCET even
159  // if is a more pessimistic evaluation.
160  // Update the delta exec.
161  task->se.exec_runtime = timer_get_ticks() - task->se.exec_start;
162  update_process_profiling_timer(task);
163 
164  // Set the sum_exec_runtime.
165  task->se.sum_exec_runtime += task->se.exec_runtime;
166 
167  // If the task is not a periodic task we have to update the virtual runtime.
168  if (!task->se.is_periodic) {
169  // Get the weight of the current task.
170  time_t weight = GET_WEIGHT(task->se.prio);
171  // If the weight is different from the default load, compute it.
172  if (weight != NICE_0_LOAD) {
173  // Get the multiplicative factor for its delta_exec.
174  double factor = ((double)NICE_0_LOAD) / ((double)weight);
175  // Weight the delta_exec with the multiplicative factor.
176  task->se.exec_runtime = (int)(((double)task->se.exec_runtime) * factor);
177  }
178  // Update vruntime of the current task.
179  task->se.vruntime += task->se.exec_runtime;
180  }
