141 static inline int elf_load_exec(task_struct *task, vfs_file_t *file, elf_header_t *hdr)
142 {
143  elf_program_header_t phdr;
144  pr_debug(" Type | Mem. Size | File Size | VADDR\n");
145  for (int idx = 0; idx < hdr->phnum; ++idx) {
146  // Get the header.
147  if (read_elf_program_header(file, hdr, idx, &phdr) == -1) {
148  pr_err("Failed to read program header at index %d.\n", idx);
149  return -1;

154  to_human_size(phdr.filesz),
155  phdr.vaddr, phdr.vaddr + phdr.memsz);
156  if (phdr.type == PT_LOAD) {
157  uint32_t virt_addr = create_vm_area(task->mm, phdr.vaddr, phdr.memsz, MM_USER | MM_RW | MM_COW, GFP_KERNEL);
158  virt_map_page_t *vpage = virt_map_alloc(phdr.memsz);
159  uint32_t dst_addr = virt_map_vaddress(task->mm, vpage, virt_addr, phdr.memsz);
160 
161  // Load the memory area.
162  vfs_read(file, (void *)dst_addr, phdr.offset, phdr.filesz);
163 
164  if (phdr.memsz > phdr.filesz) {
165  uint32_t zmem_sz = phdr.memsz - phdr.filesz;
166  memset((void *)(dst_addr + phdr.filesz), 0, zmem_sz);

168  virt_unmap_pg(vpage);
169  }
170  }
