571 static int __response_time_analysis()
572 {
573  task_struct *entry, *previous;
574  time_t r, previous_r = 0;
575  list_for_each_decl(it, &runqueue.queue)
576  {
577  entry = list_entry(it, task_struct, run_list);
578  if (entry->se.is_periodic) {
579  // Put r equal to worst case exec because is the first point in time that the task could possibly complete
580  r = entry->se.worst_case_exec;
581  previous_r = 0;
582  // The analysis can be completed either missing the deadline or reaching a fixed point
583  while (r < entry->se.deadline && r != previous_r) {
584  previous_r = r;
585  r = entry->se.worst_case_exec;
586  list_for_each_decl(it2, &runqueue.queue)
587  {
588  previous = list_entry(it2, task_struct, run_list);
589  // Check the interferences of higher priority processes
590  if (previous->se.is_periodic && previous->se.period < entry->se.period) {
591  r += (int)ceil((double)previous_r / (double)previous->se.period) * previous->se.worst_case_exec;
592  pr_debug("%d += (%.2f / %.2f) * %d\n", r, (double)previous_r, (double)previous->se.period, previous->se.worst_case_exec);
593  pr_debug("Response Time Analysis -> [%s]vs[%s] R = %d\n\n", entry->name, previous->name, r);
594  }
595  }
596  }
597  // Feasibility of scheduler is guaranteed if and only if response time analysis is lower than deadline.
598  if (r > entry->se.deadline)
599  return 1;
600  }
601  }
602  return 0;

605 int sys_waitperiod()
606 {
607  if (runqueue.curr) {
608  if (runqueue.curr->se.is_periodic) {
609  // Update the Worst Case Execution Time (WCET).
610  time_t wcet = timer_get_ticks() - runqueue.curr->se.exec_start;
611  if (runqueue.curr->se.worst_case_exec < wcet)
612  runqueue.curr->se.worst_case_exec = wcet;
613  // Update thye utilization factor.
614  runqueue.curr->se.utilization_factor = ((double)runqueue.curr->se.worst_case_exec / (double)runqueue.curr->se.period);
615 
616  // If the task is under analysis, we need to test if the process can be
617  // placed with the other periodic tasks.
618  if (runqueue.curr->se.is_under_analysis) {
619  runqueue.curr->se.worst_case_exec = runqueue.curr->se.sum_exec_runtime;
620  bool_t is_not_schedulable = false;
621 #if defined(SCHEDULER_EDF)
622  double u = 0;
623  task_struct *entry;
624  list_for_each_decl(it, &runqueue.queue)
625  {
626  entry = list_entry(it, task_struct, run_list);
627  // Sum the utilization factor of all periodic tasks.
628  if (entry->se.is_periodic)
629  u += entry->se.utilization_factor;
630  }
631  if (u > 1) {
632  is_not_schedulable = true;
633  }
634  pr_debug("utilization factor = %f\n", u);
635 #elif defined(SCHEDULER_RM)
636  // Calculating least upper bound of utilization factor.
637  // For large amount of processes ulub asymptotically should reach ln(2).
638  double ulub = (runqueue.num_periodic * (pow(2, (1.0 / runqueue.num_periodic)) - 1));
639  double u = 0;
640  task_struct *entry;
641  list_for_each_decl(it, &runqueue.queue)
642  {
643  entry = list_entry(it, task_struct, run_list);
644  // Sum the utilization factor of all periodic tasks.
645  if (entry->se.is_periodic)
646  u += entry->se.utilization_factor;
647  }
648  // If the sum of utilization factor is bounded between ulub and 1 we need to calculate
649  // the response time analysis for each process.
650  if (u > 1) {
651  is_not_schedulable = true;
652  } else if (u <= ulub)
653  is_not_schedulable = false;
654  else
655  is_not_schedulable = __response_time_analysis();
656 #endif
657  // If it is not schedulable, we need to tell it to the process.
658  if (is_not_schedulable)
659  return -ENOTSCHEDULABLE;
660 
661  // Otherwise, it is schedulable.
662  runqueue.curr->se.is_under_analysis = false;
663 
664  // The task has been executed as non-periodic process so that his deadline is not been updated
665  // by the scheduling algorithm of periodic tasks. We need to update it manually.
666  runqueue.curr->se.next_period = timer_get_ticks();
667  runqueue.curr->se.deadline = timer_get_ticks() + runqueue.curr->se.period;
668  }
669 
670  if (timer_get_ticks() > runqueue.curr->se.deadline)
671  pr_warning("%d > %d Missing deadline...\n", timer_get_ticks(), runqueue.curr->se.deadline);
672 
673  // Tell the scheduler that we have executed the periodic process.
674  runqueue.curr->se.executed = true;
675 
676  } else
677  pr_warning("An aperiodic task is calling `waitperiod`, ignoring...\n");
678  return 0;
679  }
