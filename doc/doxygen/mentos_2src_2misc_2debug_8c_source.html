93 void dbg_printf(const char *file, const char *fun, int line, char *header, const char *format, ...)
94 {
95  // Define a buffer for the formatted string.
96  static char formatted[BUFSIZ];
97  static short new_line = 1;
98 
99  // Stage 1: FORMAT
100  if (strlen(format) >= BUFSIZ)
101  return;
102 
103  // Check the log level.
104  int log_level = LOGLEVEL_DEFAULT;
105  if ((format[0] != '\0') && (format[0] == KERN_SOH_ASCII)) {
106  // Remove the Start Of Header.
107  ++format;
108  // Compute the log level.
109  log_level = (format[0] - '0');
110  // Check the log_level.
111  if ((log_level < LOGLEVEL_EMERG) || (log_level > LOGLEVEL_DEBUG))
112  log_level = 8;
113  // Remove the log_level;
114  ++format;
115  }
116  if (log_level > max_log_level) {
117  return;
118  }
119 
120  // Start variabile argument's list.
121  va_list ap;
122  va_start(ap, format);
123  // Format the message.
124  vsprintf(formatted, format, ap);
125  // End the list of arguments.
126  va_end(ap);
127 
128  // Stage 2: SEND
129  if (new_line) {
130  __debug_print_header(file, fun, line, log_level, header);
131  new_line = false;
132  }
133  for (int it = 0; (formatted[it] != 0) && (it < BUFSIZ); ++it) {
134  dbg_putchar(formatted[it]);
135  if (formatted[it] != '\n') {
136  continue;
137  }
138  if ((it + 1) >= BUFSIZ) {
139  continue;
140  }
141  if (formatted[it + 1] == 0) {
142  new_line = true;
143  } else {
144  __debug_print_header(file, fun, line, log_level, header);
145  }
146  }

173 const char *to_human_size(unsigned long bytes)
174 {
175  static char output[200];
176  const char *suffix[] = { "B", "KB", "MB", "GB", "TB" };
177  char length = sizeof(suffix) / sizeof(suffix[0]);
178  int i = 0;
179  double dblBytes = bytes;
180  if (bytes > 1024) {
181  for (i = 0; (bytes / 1024) > 0 && i < length - 1; i++, bytes /= 1024)
182  dblBytes = bytes / 1024.0;
183  }
184  sprintf(output, "%.02lf %2s", dblBytes, suffix[i]);
185  return output;
186 }
