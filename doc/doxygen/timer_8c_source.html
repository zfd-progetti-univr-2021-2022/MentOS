347 void run_timer_softirq()
348 {
349  tvec_base_t *base = &cpu_base;
350  spinlock_lock(&base->lock);
351 
352 #ifdef ENABLE_REAL_TIMER_SYSTEM
353 
354  // While we are not up to date with current ticks
355  unsigned long current_ticks = timer_get_ticks();
356  while (base->timer_ticks <= current_ticks) {
357 
358  // Index of the current timer to execute
359  int current_time_index = base->timer_ticks & TVR_MASK;
360 
361  // If the index is zero then all lists in base->tv1 have been checked, so they are empty
362  if (!current_time_index) {
363 
364  // Consider the first invocation of the cascade() function: it receives as arguments
365  // the address in base, the address of base->tv2, and the index of the list
366  // in base->tv2 including the timers that will decay in the next 256 ticks. This
367  // index is determined by looking at the proper bits of the base->timer_ticks value.
368  // cascade() moves all dynamic timers in the base->tv2 list into the
369  // proper lists of base->tv1; then, it returns a positive value, unless all base->tv2
370  // lists are now empty. If so, cascade() is invoked once more to replenish
371  // base->tv2 with the timers included in a list of base->tv3, and so on.
372 
373  int tv2_index = (base->timer_ticks >> TIMER_TICKS_BITS(0)) & TVN_MASK;
374  int tv3_index = (base->timer_ticks >> TIMER_TICKS_BITS(1)) & TVN_MASK;
375  int tv4_index = (base->timer_ticks >> TIMER_TICKS_BITS(2)) & TVN_MASK;
376  int tv5_index = (base->timer_ticks >> TIMER_TICKS_BITS(3)) & TVN_MASK;
377 
378  if (!cascate(base, &base->tv2, tv2_index, 2) &&
379  !cascate(base, &base->tv3, tv3_index, 3) &&
380  !cascate(base, &base->tv4, tv4_index, 4) &&
381  !cascate(base, &base->tv5, tv5_index, 5));
382  }
383 
384  // If there are timers to execute in this instant
385  if (!list_head_empty(&base->tv1.vec[current_time_index])) {
386  pr_notice("Executing dynamic timers at %d ticks from start inside of tv1.vec[%d]\n",
387  base->timer_ticks, current_time_index);
388 
389  // Trigger all timers
390  struct list_head *it, *tmp;
391  list_for_each_safe (it, tmp, &base->tv1.vec[current_time_index]) {
392  struct timer_list *timer = list_entry(it, struct timer_list, entry);
393 
394  // Executes timer function
395  spinlock_unlock(&base->lock);
396  pr_notice("Executing dynamic timer function...\n");
397  timer->function(timer->data);
398  spinlock_lock(&base->lock);
399 
400  // Removes timer from list
401  list_head_del(it);
402  kfree(timer);
403  }
404  }
405 
406  // Advance timer check
407  ++base->timer_ticks;
408  }
409 
410  base->running_timer = NULL;
411 
412 #else
413 
414  struct list_head *it, *tmp;
415  list_for_each_safe (it, tmp, &base->list) {
416  struct timer_list *timer = list_entry(it, struct timer_list, entry);
417  if (timer->expires <= timer_get_ticks()) {
418  base->running_timer = timer;
419  timer->base = NULL;
420 
421  // Executes timer function
422  spinlock_unlock(&base->lock);
423  pr_notice("Executing dynamic timer function...\n");
424  timer->function(timer->data);
425  spinlock_lock(&base->lock);
426 
427  // Removes timer from list
428  pr_notice("Removing dynamic timer...\n");
429  list_head_del(it);
430  kfree(timer);
431  }
432  }
433 
434 #endif
435 
436  base->running_timer = NULL;
437  spinlock_unlock(&base->lock);
438 }
