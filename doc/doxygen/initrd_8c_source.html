450 static ssize_t initrd_read(vfs_file_t *file, char *buf, off_t offset, size_t nbyte)
451 {
452  // If the number of byte to read is zero, skip.
453  if (nbyte == 0) {
454  return 0;
455  }
456  // Get the file descriptor of the file.
457  int lfd = file->ino;
458  // Get the current position.
459  int read_pos = offset;
460  // Get the length of the file.
461  int file_size = (int)fs_specs.headers[lfd].length;
462  // If we have reached the end of the file, return.
463  if (read_pos == file_size) {
464  return EOF;
465  }
466  // Get the begin of the file.
467  char *file_start = (char *)(modules[0].mod_start + fs_specs.headers[lfd].offset);
468  // Declare an iterator, used afterward to return the number of bytes read.
469  ssize_t it = 0;
470  while ((it < nbyte) && (read_pos < file_size)) {
471  *buf++ = file_start[read_pos];
472  ++read_pos;
473  ++it;
474  }
475  return it;
476 }
