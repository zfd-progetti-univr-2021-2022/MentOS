23 static ssize_t procv_read(vfs_file_t *file, char *buf, off_t offset, size_t nbyte)
24 {
25  if (buf == NULL) {
26  return -1;
27  }
28  // Read the character from the keyboard.
29  int c = keyboard_getc();
30  if (c >= 0) {
31  // Return the character.
32  *((char *)buf) = c;
33  if (bitmask_check(ktermios.c_lflag, ECHO)) {
34  if ((c == '\b') &&
35  (!bitmask_check(ktermios.c_lflag, ICANON) || !bitmask_check(ktermios.c_lflag, ECHOE))) {
36  return 1;
37  }
38  // Echo the character to video.
39  video_putc(c);
40  }
41  return 1;
42  }
43  return 0;
44 }

46 static ssize_t procv_write(vfs_file_t *file, const void *buf, off_t offset, size_t nbyte)
