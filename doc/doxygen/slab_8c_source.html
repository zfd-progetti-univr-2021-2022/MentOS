38 static int __alloc_slab_page(kmem_cache_t *cachep, gfp_t flags)
39 {
40  page_t *page = _alloc_pages(flags, cachep->gfp_order);
41  if (!page) {
42  pr_crit("Failed to allocate a new page from slab.\n");
43  return -1;
44  }
45 
46  list_head_init(&page->slabs);
47 
48  // Save in the root page the kmem_cache_t pointer,
49  // to allow freeing arbitrary pointers
50  page[0].container.slab_cache = cachep;
51 
52  // Update slab main pages of all child pages, to allow
53  // reconstructing which page handles a specified address
54  for (unsigned int i = 1; i < (1U << cachep->gfp_order); i++) {
55  page[i].container.slab_main_page = page;
56  }
57 
58  unsigned int slab_size = PAGE_SIZE * (1U << cachep->gfp_order);
59 
60  // Update the page objects counters
61  page->slab_objcnt = slab_size / cachep->size;
62  page->slab_objfree = page->slab_objcnt;
63 
64  unsigned int pg_addr = get_lowmem_address_from_page(page);
65 
66  list_head_init(&page->slab_freelist);
67 
68  // Build the objects structures
69  for (unsigned int i = 0; i < page->slab_objcnt; i++) {
70  kmem_obj *obj = KMEM_OBJ(cachep, pg_addr + cachep->size * i);
71  list_head_add(&obj->objlist, &page->slab_freelist);
72  }
73 
74  // Add the page to the slab list and update the counters
75  list_head_add(&page->slabs, &cachep->slabs_free);
76  cachep->total_num += page->slab_objcnt;
77  cachep->free_num += page->slab_objcnt;
78 
79  return 0;
80 }

97 static void __compute_size_and_order(kmem_cache_t *cachep)
98 {
99  // Align the whole object to the required padding
100  cachep->size = __find_next_alignment(
101  max(cachep->object_size, KMEM_OBJ_OVERHEAD),
102  max(8, cachep->align));
103 
104  // Compute the gfp order
105  unsigned int size = __find_next_alignment(cachep->size, PAGE_SIZE) / PAGE_SIZE;
106  while ((size /= 2) > 0) {
107  cachep->gfp_order++;
108  }
109 }

135 static inline void *__kmem_cache_alloc_slab(kmem_cache_t *cachep, page_t *slab_page)
136 {
137  list_head *elem_listp = list_head_pop(&slab_page->slab_freelist);
138  if (!elem_listp) {
139  pr_warning("There are no FREE element inside the slab_freelist\n");
140  return NULL;
141  }
142  slab_page->slab_objfree--;
143  cachep->free_num--;
144 
145  kmem_obj *obj = list_entry(elem_listp, kmem_obj, objlist);
146 
147  // Get the element from the kmem_obj object
148  void *elem = ADDR_FROM_KMEM_OBJ(cachep, obj);
149 
150  if (cachep->ctor)
151  cachep->ctor(elem);
152 
153  return elem;
154 }

156 static inline void __kmem_cache_free_slab(kmem_cache_t *cachep, page_t *slab_page)
157 {
158  cachep->free_num -= slab_page->slab_objfree;
159  cachep->total_num -= slab_page->slab_objcnt;
160  // Clear objcnt, used as a flag to check if the page belongs to the slab
161  slab_page->slab_objcnt = 0;
162  slab_page->container.slab_main_page = NULL;
163 
164  // Reset all non-root slab pages
165  for (unsigned int i = 1; i < (1U << cachep->gfp_order); i++) {
166  (slab_page + i)->container.slab_main_page = NULL;
167  }
168 
169  __free_pages(slab_page);
170 }

207 void kmem_cache_destroy(kmem_cache_t *cachep)
208 {
209  while (!list_head_empty(&cachep->slabs_free)) {
210  list_head *slab_list = list_head_pop(&cachep->slabs_free);
211  __kmem_cache_free_slab(cachep, list_entry(slab_list, page_t, slabs));
212  }
213 
214  while (!list_head_empty(&cachep->slabs_partial)) {
215  list_head *slab_list = list_head_pop(&cachep->slabs_partial);
216  __kmem_cache_free_slab(cachep, list_entry(slab_list, page_t, slabs));
217  }
218 
219  while (!list_head_empty(&cachep->slabs_full)) {
220  list_head *slab_list = list_head_pop(&cachep->slabs_full);
221  __kmem_cache_free_slab(cachep, list_entry(slab_list, page_t, slabs));
222  }
223 
224  kmem_cache_free(cachep);
225  list_head_del(&cachep->cache_list);
226 }

228 #ifdef ENABLE_CACHE_TRACE
229 void *pr_kmem_cache_alloc(const char *file, const char *fun, int line, kmem_cache_t *cachep, gfp_t flags)
230 #else
231 void *kmem_cache_alloc(kmem_cache_t *cachep, gfp_t flags)
232 #endif
233 {
234  if (list_head_empty(&cachep->slabs_partial)) {
235  if (list_head_empty(&cachep->slabs_free)) {
236  if (flags == 0)
237  flags = cachep->flags;
238 
239  // Refill the cache in an exponential fashion, capping at KMEM_MAX_REFILL_OBJ_COUNT to avoid
240  // too big allocations
241  __kmem_cache_refill(cachep, min(cachep->total_num, KMEM_MAX_REFILL_OBJ_COUNT), flags);
242  if (list_head_empty(&cachep->slabs_free)) {
243  pr_crit("Cannot allocate more slabs in `%s`\n", cachep->name);
244  return NULL;
245  }
246  }
247 
248  // Add a free slab to partial list because in any case an element will
249  // be removed before the function returns
250  list_head *free_slab = list_head_pop(&cachep->slabs_free);
251  list_head_add(free_slab, &cachep->slabs_partial);
252  }
253 
254  page_t *slab_page = list_entry(list_head_front(&cachep->slabs_partial), page_t, slabs);
255  void *ptr = __kmem_cache_alloc_slab(cachep, slab_page);
256 
257  // If the slab is now full, add it to the full slabs list
258  if (slab_page->slab_objfree == 0) {
259  list_head *slab_full_elem = list_head_pop(&cachep->slabs_partial);
260  list_head_add(slab_full_elem, &cachep->slabs_full);
261  }
262 #ifdef ENABLE_CACHE_TRACE
263  pr_notice("kmem_cache_alloc : (%-16s:%3d)[%-16s] : 0x%p\n", file, line, cachep->name, ptr);
264 #endif
265  return ptr;
266 }

269 void pr_kmem_cache_free(const char *file, const char *fun, int line, void *ptr)
270 #else
271 void kmem_cache_free(void *ptr)
272 #endif
273 {
274  page_t *slab_page = get_lowmem_page_from_address((uint32_t)ptr);
275 
276  // If the slab main page is a lowmem page, change to it as it's the root page
277  if (is_lowmem_page_struct(slab_page->container.slab_main_page)) {
278  slab_page = slab_page->container.slab_main_page;
279  }
280 
281  kmem_cache_t *cachep = slab_page->container.slab_cache;
282 
283 #ifdef ENABLE_CACHE_TRACE
284  pr_notice("kmem_cache_free : (%-16s:%3d)[%-16s] : 0x%p\n", file, line, cachep->name, ptr);
285 #endif
286  if (cachep->dtor)
287  cachep->dtor(ptr);
288 
289  kmem_obj *obj = KMEM_OBJ(cachep, ptr);
290 
291  // Add object to the free list
292  list_head_add(&obj->objlist, &slab_page->slab_freelist);
293  slab_page->slab_objfree++;
294  cachep->free_num++;
295 
296  // Now page is completely free
297  if (slab_page->slab_objfree == slab_page->slab_objcnt) {
298  // Remove page from partial list
299  list_head_del(&slab_page->slabs);
300  // Add page to free list
301  list_head_add(&slab_page->slabs, &cachep->slabs_free);
302  }
303  // Now page is not full, so change its list
304  else if (slab_page->slab_objfree == 1) {
305  // Remove page from full list
306  list_head_del(&slab_page->slabs);
307  // Add page to partial list
308  list_head_add(&slab_page->slabs, &cachep->slabs_partial);
309  }
310 }
