34 static inline int day_of_week(unsigned int y, unsigned int m, unsigned int d)
35 {
36  int h, j, k;
37  // January and February are counted as months 13 and 14 of the previous year
38  if (m <= 2) {
39  m += 12;
40  y -= 1;
41  }
42  // J is the century
43  j = (int)(y / 100);
44  // K the year of the century
45  k = (int)(y % 100);
46  // Compute H using Zeller's congruence
47  h = (int)(d + (26 * (m + 1) / 10) + k + (k / 4) + (5 * j) + (j / 4));
48  // Return the day of the week
49  return ((h + 5) % 7) + 1;
50 }
