293  // Get what's left.
294  uint32_t remaining_size = chunk_size - block_size;
295  // Get the real size.
296  uint32_t real_size = (remaining_size < (8 + OVERHEAD)) ? chunk_size : block_size;
297  // Set the size of the best fitting block.

325

331  // goto no_split;
332  }

334  blkmngr_add_to_freelist(t, freelist);
335  } else {
336  // Choice a) seperate!
337  block_t *putThisBack = (block_t *)block_ptr;
338  putThisBack->size = remaining_size - OVERHEAD;
339  blkmngr_set_free(&(putThisBack->size), 1);
340 
341  putThisBack->next = stored_next_block;
342 
343  if (base_ptr == tail_block) {

346  // ksbrk(-reclaimSize);
347  // goto no_split;
348  }
349  blkmngr_add_to_freelist(putThisBack, freelist);
350  }
351

353  }

355  // Remove the block from the free list.
356  blkmngr_remove_from_freelist(base_ptr, freelist);
357 
358  return (char *)base_ptr + sizeof(block_t);
359  } else {
360  uint32_t realsize = block_size;
361  block_t *ret;
362 
363  if (heap == &kernel_heap) {

375  }

385  */

390  // ptr = ptr + block_size - sizeof(uint32_t);
391  // trailing_space = ptr;
392  // *trailing_space = ret->size;
393 
394  // Now, return it!

665 static void __do_free(vm_area_struct_t *heap, void *ptr)
666 {
667  assert(ptr);
668 
669  // Get:
670  // 1) First memory block.
671  // static block_t *head = NULL;
672  // 2) Last memory block.
673  // static block_t *tail = NULL;
674  // 3) All the memory blocks that are freed.
675  // static block_t *freelist = NULL;
676 
677  // We will use these in writing.
678  uint32_t *head = (uint32_t *)(heap->vm_start);
679  uint32_t *tail = (uint32_t *)(heap->vm_start + sizeof(block_t *));
680  uint32_t *freelist = (uint32_t *)(heap->vm_start + 2 * sizeof(block_t *));
681  assert(head && tail && freelist && "Heap block lists point to null.");
682 
683  // We will use these others in reading.
684  block_t *tail_block = (block_t *)*tail;
685 
686  block_t *curr = (block_t *)((char *)ptr - sizeof(block_t));
687 
688  block_t *prev = blkmngr_get_previous_block(curr, head);
689  block_t *next = blkmngr_get_next_block(curr, tail);
690  if (blkmngr_is_free(prev) && blkmngr_is_free(next)) {
691  prev->size =
692  blkmngr_get_real_size(prev->size) + 2 * OVERHEAD +
693  blkmngr_get_real_size(curr->size) +
694  blkmngr_get_real_size(next->size);
695  blkmngr_set_free(&(prev->size), 1);
696 
697  prev->next = blkmngr_get_next_block(next, tail);
698 
699  // If next used to be tail, set prev = tail.
700  if (tail_block == next) {
701  *tail = (uint32_t)prev;
702  }
703  blkmngr_remove_from_freelist(next, freelist);
704  } else if (blkmngr_is_free(prev)) {
705  prev->size =
706  blkmngr_get_real_size(prev->size) + OVERHEAD + blkmngr_get_real_size(curr->size);
707  blkmngr_set_free(&(prev->size), 1);
708 
709  prev->next = next;
710 
711  if (tail_block == curr) {
712  *tail = (uint32_t)prev;
713  }
714  } else if (blkmngr_is_free(next)) {
715  // Change size to curr's size + OVERHEAD + next's size.
716  curr->size =
717  blkmngr_get_real_size(curr->size) + OVERHEAD + blkmngr_get_real_size(next->size);
718  blkmngr_set_free(&(curr->size), 1);
719 
720  curr->next = blkmngr_get_next_block(next, tail);
721 
722  if (tail_block == next) {
723  *tail = (uint32_t)curr;
724  }
725  blkmngr_remove_from_freelist(next, freelist);
726  blkmngr_add_to_freelist(curr, freelist);
727  } else {
728  // Just mark curr freed.
729  blkmngr_set_free(&(curr->size), 1);
730  blkmngr_add_to_freelist(curr, freelist);
731  }
732 }

775 void *sys_brk(void *addr)
776 {
777  // Get user heap segment structure.
778  vm_area_struct_t *heap_segment = __find_user_heap();
779 
780  // Allocate the segment if don't exist.
781  if (heap_segment == NULL) {
782  task_struct *current_task = scheduler_get_current_process();
783  mm_struct_t *current_mm = current_task->mm;
784  current_mm->start_brk = create_vm_area(current_mm,
785  0x40000000 /*FIXME! stabilize this*/,
786  UHEAP_INITIAL_SIZE, MM_RW | MM_PRESENT | MM_USER | MM_UPDADDR, GFP_HIGHUSER);
787  current_mm->brk = current_mm->start_brk;
788  // Reserved space for:
789  // 1) First memory block.
790  // static block_t *head = NULL;
791  // 2) Last memory block.
792  // static block_t *tail = NULL;
793  // 3) All the memory blocks that are freed.
794  // static block_t *freelist = NULL;
795  current_mm->brk += 3 * sizeof(block_t *);
796  heap_segment = __find_user_heap();
797  }
798  // If the address falls inside the memory region, call the free function,
799  // otherwise execute a malloc of the specified amount.
800  if (((uintptr_t)addr > heap_segment->vm_start) &&
801  ((uintptr_t)addr < heap_segment->vm_end)) {
802  __do_free(heap_segment, addr);
803  return NULL;
804  }
805  return __do_malloc(heap_segment, (uintptr_t)addr);
806 }
