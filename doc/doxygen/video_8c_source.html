264 {

266  // Move the upper buffer up by one line.
267  for (int row = 0; row < (STORED_PAGES * HEIGHT); ++row) {
268  memcpy(upper_buffer + W2 * row, upper_buffer + W2 * (row + 1), 2 * WIDTH);

270  // Copy the first line on the screen inside the last line of the upper buffer.

272  // Move the screen up by one line.
