180 static inline void __collect_signal(int sig, sigpending_t *list, siginfo_t *info)
181 {
182  pr_debug("__collect_signal(%d, %p, %p)\n", sig, list, info);
183  assert(list && "Null `list` structure.");
184  assert(info && "Null `info` structure.");
185 
186  sigqueue_t *queue_entry = NULL;
187  bool_t still_pending = false;
188  // Collect the siginfo appropriate to this signal. Check if
189  // there is another siginfo for the same signal.
190  list_for_each_decl(it, &list->list)
191  {
192  sigqueue_t *q = list_entry(it, sigqueue_t, list);
193  pr_debug("__collect_signal(%d, %p, %p) : Signal in queue : %p(%d : %s).\n", sig, list, info,
194  q, q->info.si_signo, strsignal(q->info.si_signo));
195  if (q->info.si_signo == sig) {
196  // If the entry is already set, this means that there are several handlers
197  // pending for this particular signal.
198  if (queue_entry) {
199  pr_debug("__collect_signal(%d, %p, %p) : Still pending, do not remove from set.\n", sig, list, info);
200  still_pending = true;
201  break;
202  }
203  // Store the entry we encounter.
204  queue_entry = q;
205  }
206  }
207  // If there are no other signals pending of the same type,
208  // remove the signal from the set.
209  if (!still_pending) {
210  sigdelset(&list->signal, sig);
211  pr_debug("__collect_signal(%d, %p, %p) : Remove signal from set: %d.\n", sig, list, info,
212  list->signal.sig[0]);
213  }
214  // If we have found an entry.
215  if (queue_entry) {
216  pr_debug("__collect_signal(%d, %p, %p) : Remove and delete sigqueue entry : %p.\n", sig, list, info, queue_entry);
217  // Remove the entry from the queue.
218  list_head_del(&queue_entry->list);
219  // Copy the details about the entry inside the info structure.
220  __copy_siginfo(info, &queue_entry->info);
221  // Free the memory for the queue entry.
222  __sigqueue_free(queue_entry);
223  } else {
224  pr_debug("__collect_signal(%d, %p, %p) : Cannot find the signal in the queue.\n", sig, list, info);
225  // Ok, it wasn't in the queue, zero out the info.
226  __clear_siginfo(info);
227  // Get the current process.
228  struct task_struct *current = scheduler_get_current_process();
229  assert(current && "There is no running process.");
230  // Initialize the info.
231  info->si_signo = sig;
232  info->si_code = SI_USER;
233  info->si_value.sival_int = 0;
234  info->si_errno = 0;
235  info->si_pid = current->pid;
236  info->si_uid = current->uid;
237  info->si_addr = NULL;
238  info->si_status = 0;
239  info->si_band = 0;
240  }
241 }

256 static inline int __handle_signal(int signr, siginfo_t *info, sigaction_t *ka, struct pt_regs *regs)
257 {
258  pr_debug("__handle_signal(%d, %p, %p, %p)\n", signr, info, ka, regs);
259  // The do_signal() function is usually only invoked when the CPU is going
260  // to return in User Mode.
261  struct task_struct *current = scheduler_get_current_process();
262  assert(current && "There is no running process.");
263  // Skip the `init` process, always.
264  if (current->pid == 1) {
265  errno = ESRCH;
266  return 0;
267  }
268  // Save the previous signal mask.
269  memcpy(&current->saved_sigmask, &current->blocked, sizeof(sigset_t));
270 
271  // Add the signal to the list of blocked signals.
272  sigaddset(&current->blocked, signr);
273 
274  // Store the registers before setting the ones required by the signal handling.
275  current->thread.signal_regs = *regs;
276 
277  // Restore the registers for the process that has set the signal.
278  *regs = current->thread.regs;
279 
280  // Set the instruction pointer.
281  regs->eip = (uintptr_t)ka->sa_handler;
282 
283  // If the user is also asking for the signal info, push it into the stack.
284  if (bitmask_check(ka->sa_flags, SA_SIGINFO)) {
285  // Move the stack so that we have space for storing the siginfo.
286  regs->useresp -= sizeof(siginfo_t);
287  // Save the pointer where the siginfo is stored.
288  siginfo_t *siginfo_addr = (siginfo_t *)regs->useresp;
289  // We push on the stack the entire siginfo.
290  __copy_siginfo(siginfo_addr, info);
291  // We push on the stack the pointer to the siginfo we copied on the stack.
292  PUSH_ARG(regs->useresp, siginfo_t *, siginfo_addr);
293  }
294 
295  // Push on the stack the signal number, first and only argument of the handler.
296  PUSH_ARG(regs->useresp, int, signr);
297 
298  // Push on the stack the function required to handle the signal return.
299  PUSH_ARG(regs->useresp, uint32_t, current->sigreturn_eip);
300 
301  return 1;
302 }

370 int do_signal(struct pt_regs *f)
371 {
372  // The do_signal() function is usually only invoked when the CPU is going
373  // to return in User Mode.
374  struct task_struct *current = scheduler_get_current_process();
375  if (current == NULL)
376  return 0;
377 
378  // First, checks whether the function itself was triggered by an interrupt;
379  // if so, it simply returns. Otherwise, if the function was triggered by an
380  // exception that was raised while the process was running in User Mode,
381  // the function continues executing.
382  if ((f->cs & 3) != 3)
383  return 0;
384 
385  // Create a siginfo.
386  siginfo_t info;
387  // The return code of __dequeue_signal( ) is stored in signr.
388  int signr, exit_code;
389 
390  // Lock the signal handling for the given task.
391  __lock_task_sighand(current);
392 
393  // The heart of the do_signal( ) function consists of a loop that
394  // repeatedly invokes the __dequeue_signal( ) function until no
395  // non-blocked pending signals are left.
396  while (!list_head_empty(&current->pending.list)) {
397  // Get the signal to deliver.
398  signr = exit_code = __dequeue_signal(&current->pending, &current->blocked, &info);
399 
400  // Check the signal that we want to send.
401  if ((signr < 0) || (signr >= NSIG)) {
402  pr_err("Wrong signal number!\n");
403  break;
404  }
405 
406  // If its value is 0, it means that all pending signals have been
407  // handled and do_signal( ) can finish.
408  if (signr == 0) {
409  pr_notice("There are no more signals to handle.\n");
410  __unlock_task_sighand(current);
411  return 0;
412  }
413 
414  // Get the associated signal action.
415  sigaction_t *ka = &current->sighand.action[signr - 1];
416 
417  // The only exception comes when the receiving process is init, in
418  // which case the signal is discarded.
419  if (current->pid == 1)
420  continue;
421 
422  // When a delivered signal is explicitly ignored, the do_signal( )
423  // function normally just continues with a new execution of the loop
424  // and therefore considers another pending signal.
425  if (ka->sa_handler == SIG_IGN) {
426  if (signr == SIGCHLD)
427  while (sys_waitpid(-1, NULL, WNOHANG) > 0) {}
428  continue;
429  }
430 
431  // When a delivered signal is the default one, do_signal( ) must
432  // perform the default action of the signal.
433  if (ka->sa_handler == SIG_DFL) {
434  // For other processes, since the default action depends on the
435  // type of signal, the function executes a switch statement based
436  // on the value of signr.
437  switch (signr) {
438  // The signals whose default action is "ignore" are easily handled:
439  case SIGCONT:
440  case SIGCHLD:
441  case SIGURG:
442  case SIGWINCH:
443  continue;
444  // The signals whose default action is "stop" may stop the
445  // current process. To do this, do_signal( ) sets the state
446  // of current to TASK_STOPPED and then invokes the schedule( )
447  // function (see Section 11.2.2).
448  // The difference between SIGSTOP and the other signals is:
449  // SIGSTOP always stops the process;
450  // The other signals stop the process only if it is not
451  // in an "orphaned process group."
452  case SIGTSTP:
453  case SIGTTIN:
454  case SIGTTOU:
455  if (is_orphaned_pgrp(current->gid))
456  continue;
457 
458  case SIGSTOP:
459  __unlock_task_sighand(current);
460  __do_signal_stop(current, f, signr);
461  __lock_task_sighand(current);
462 
463  continue;
464  case SIGQUIT:
465  case SIGILL:
466  case SIGTRAP:
467 
468  case SIGABRT:
469  sys_exit(3);
470 
471  continue;
472  case SIGFPE:
473  case SIGSEGV:
474  case SIGBUS:
475  case SIGSYS:
476  case SIGXCPU:
477  case SIGXFSZ:
478 #if 0
479  if (do_coredump(signr, f))
480  exit_code |= 0x80;
481 #endif
482  default:
483 #if 0
484  current->flags |= PF_SIGNALED;
485 #endif
486  sys_exit(exit_code);
487  __unlock_task_sighand(current);
488  return 1;
489  }
490  }
491  if (__handle_signal(signr, &info, ka, f) == 1) {
492  __unlock_task_sighand(current);
493  return 1;
494  }
495  pr_emerg("Failed to handle signal.\n");
496  }
497  __unlock_task_sighand(current);
498  return 0;
499 }

671 int sys_sigaction(int signum, const sigaction_t *act, sigaction_t *oldact)
672 {
673  pr_debug("sys_sigaction(%d, %p, %p)\n", signum, act, oldact);
674  // Check the signal that we want to send.
675  if ((signum < 0) || (signum >= NSIG)) {
676  pr_debug("sys_sigaction(%d, %p, %p): Wrong signal number!\n", signum, act, oldact);
677  return -EINVAL;
678  }
679  // The do_signal() function is usually only invoked when the CPU is going
680  // to return in User Mode.
681  struct task_struct *current = scheduler_get_current_process();
682  assert(current && "There is no running process.");
683  // Skip the `init` process, always.
684  if (current->pid == 1) {
685  pr_debug("sys_sigaction(%d, %p, %p): Cannot set signal for init!\n", signum, act, oldact);
686  return -EINVAL;
687  }
688  // Lock the signal handling for the given task.
689  __lock_task_sighand(current);
690  // Get a pointer to the entry in the sighand.action array.
691  sigaction_t *current_sigaction = &current->sighand.action[signum - 1];
692  pr_debug("sys_sigaction(%d, %p, %p): : Signal old action ptr %p\n", signum, act, oldact, current_sigaction);
693  // If requested, get the old sigaction.
694  if (oldact) {
695  memcpy(oldact, current_sigaction, sizeof(sigaction_t));
696  }
697  // Set the new action.
698  memcpy(current_sigaction, act, sizeof(sigaction_t));
699  // Unlock the signal handling for the given task.
700  __unlock_task_sighand(current);
701  // Return the old sighandler.
702  return 0;
703 }

705 int sys_sigprocmask(int how, const sigset_t *set, sigset_t *oldset)
706 {
707  pr_notice("sys_sigprocmask(%d, %p, %p)\n", how, set, oldset);
708  if (!set && !oldset) {
709  return -EFAULT;
710  }
711  if ((how < SIG_BLOCK) || (how > SIG_SETMASK)) {
712  return -EINVAL;
713  }
714  // The do_signal() function is usually only invoked when the CPU is going
715  // to return in User Mode.
716  struct task_struct *current = scheduler_get_current_process();
717  assert(current && "There is no running process.");
718  // Skip the `init` process, always.
719  if (current->pid == 1) {
720  pr_notice("sys_sigprocmask(%d, %p, %p): Cannot set signal for init!\n", how, set, oldset);
721  return -EINVAL;
722  }
723  // If `oldset` is not, return the old set.
724  if (oldset) {
725  oldset->sig[0] = current->blocked.sig[0];
726  oldset->sig[1] = current->blocked.sig[1];
727  }
728  // Set the new signal mask.
729  if (set) {
730  if (how == SIG_BLOCK) {
731  // The set of blocked signals is the union of the current set
732  // and the set argument.
733  current->blocked.sig[0] |= set->sig[0];
734  current->blocked.sig[1] |= set->sig[1];
735  } else if (how == SIG_UNBLOCK) {
736  // The signals in set are removed from the current set of
737  // blocked signals. It is permissible to attempt to unblock
738  // a signal which is not blocked.
739  current->blocked.sig[0] &= ~(set->sig[0]);
740  current->blocked.sig[1] &= ~(set->sig[1]);
741  } else if (how == SIG_SETMASK) {
742  // The set of blocked signals is set to the argument set.
743  current->blocked.sig[0] = set->sig[0];
744  current->blocked.sig[1] = set->sig[1];
745  }
746  }
747  return 0;
748 }
