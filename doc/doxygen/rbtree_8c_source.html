225 int rbtree_tree_insert_node(rbtree_t *tree, rbtree_node_t *node)
226 {
227  if (tree && node) {
228  if (tree->root == NULL) {
229  tree->root = node;
230  } else {
231  rbtree_node_t head = { 0 }; // False tree root
232  rbtree_node_t *g, *t; // Grandparent & parent
233  rbtree_node_t *p, *q; // Iterator & parent
234  int dir = 0, last = 0;
235 
236  // Set up our helpers
237  t = &head;
238  g = p = NULL;
239  q = t->link[1] = tree->root;
240 
241  // Search down the tree for a place to insert
242  while (1) {
243  if (q == NULL) {
244  // Insert node at the first null link.
245  p->link[dir] = q = node;
246  } else if (rbtree_node_is_red(q->link[0]) &&
247  rbtree_node_is_red(q->link[1])) {
248  // Simple red violation: color flip
249  q->red = 1;
250  q->link[0]->red = 0;
251  q->link[1]->red = 0;
252  }
253 
254  if (rbtree_node_is_red(q) && rbtree_node_is_red(p)) {
255  // Hard red violation: rotations necessary
256  int dir2 = t->link[1] == g;
257  if (q == p->link[last]) {
258  t->link[dir2] = rbtree_node_rotate(g, !last);
259  } else {
260  t->link[dir2] = rbtree_node_rotate2(g, !last);
261  }
262  }
263 
264  // Stop working if we inserted a node. This
265  // check also disallows duplicates in the tree
266  if (tree->cmp(tree, q, node) == 0) {
267  break;
268  }
269 
270  last = dir;
271  dir = tree->cmp(tree, q, node) < 0;
272 
273  // Move the helpers down
274  if (g != NULL) {
275  t = g;
276  }
277 
278  g = p, p = q;
279  q = q->link[dir];
280  }
281 
282  // Update the root (it may be different)
283  tree->root = head.link[1];
284  }
285 
286  // Make the root black for simplified logic
287  tree->root->red = 0;
288  ++tree->size;
289  }
290 
291  return 1;
292 }

297 int rbtree_tree_remove_with_cb(rbtree_t *tree,
298  void *value,
299  rbtree_tree_node_f node_cb)
300 {
301  if (tree->root != NULL) {
302  rbtree_node_t head = { 0 }; // False tree root
303  rbtree_node_t node = { .value = value }; // Value wrapper node
304  rbtree_node_t *q, *p, *g; // Helpers
305  rbtree_node_t *f = NULL; // Found item
306  int dir = 1;
307 
308  // Set up our helpers
309  q = &head;
310  g = p = NULL;
311  q->link[1] = tree->root;
312 
313  // Search and push a red node down
314  // to fix red violations as we go
315  while (q->link[dir] != NULL) {
316  int last = dir;
317 
318  // Move the helpers down
319  g = p, p = q;
320  q = q->link[dir];
321  dir = tree->cmp(tree, q, &node) < 0;
322 
323  // Save the node with matching value and keep
324  // going; we'll do removal tasks at the end
325  if (tree->cmp(tree, q, &node) == 0) {
326  f = q;
327  }
328 
329  // Push the red node down with rotations and color flips
330  if (!rbtree_node_is_red(q) && !rbtree_node_is_red(q->link[dir])) {
331  if (rbtree_node_is_red(q->link[!dir])) {
332  p = p->link[last] = rbtree_node_rotate(q, dir);
333  } else if (!rbtree_node_is_red(q->link[!dir])) {
334  rbtree_node_t *s = p->link[!last];
335  if (s) {
336  if (!rbtree_node_is_red(s->link[!last]) &&
337  !rbtree_node_is_red(s->link[last])) {
338  // Color flip
339  p->red = 0;
340  s->red = 1;
341  q->red = 1;
342  } else {
343  int dir2 = g->link[1] == p;
344  if (rbtree_node_is_red(s->link[last])) {
345  g->link[dir2] = rbtree_node_rotate2(p, last);
346  } else if (rbtree_node_is_red(s->link[!last])) {
347  g->link[dir2] = rbtree_node_rotate(p, last);
348  }
349 
350  // Ensure correct coloring
351  q->red = g->link[dir2]->red = 1;
352  g->link[dir2]->link[0]->red = 0;
353  g->link[dir2]->link[1]->red = 0;
354  }
355  }
356  }
357  }
358  }
359 
360  // Replace and remove the saved node
361  if (f) {
362  void *tmp = f->value;
363  f->value = q->value;
364  q->value = tmp;
365 
366  p->link[p->link[1] == q] = q->link[q->link[0] == NULL];
367 
368  if (node_cb) {
369  node_cb(tree, q);
370  }
371  q = NULL;
372  }
373 
374  // Update the root (it may be different)
375  tree->root = head.link[1];
376 
377  // Make the root black for simplified logic
378  if (tree->root != NULL) {
379  tree->root->red = 0;
380  }
381 
382  --tree->size;
383  }
384  return 1;
385 }

458 // Traverse a red black tree in the user-specified direction (0 asc, 1 desc)
459 static void *rbtree_iter_move(rbtree_iter_t *iter, int dir)
460 {
461  if (iter->node->link[dir] != NULL) {
462  // Continue down this branch
463  iter->path[iter->top++] = iter->node;
464  iter->node = iter->node->link[dir];
465  while (iter->node->link[!dir] != NULL) {
466  iter->path[iter->top++] = iter->node;
467  iter->node = iter->node->link[!dir];
468  }
469  } else {
470  // Move to the next branch
471  rbtree_node_t *last = NULL;
472  do {
473  if (iter->top == 0) {
474  iter->node = NULL;
475  break;
476  }
477  last = iter->node;
478  iter->node = iter->path[--iter->top];
479  } while (last == iter->node->link[dir]);
480  }
481  return iter->node == NULL ? NULL : iter->node->value;
482 }

504 int rbtree_tree_test(rbtree_t *tree, rbtree_node_t *root)
505 {
506  int lh, rh;
507 
508  if (root == NULL)
509  return 1;
510  else {
511  rbtree_node_t *ln = root->link[0];
512  rbtree_node_t *rn = root->link[1];
513 
514  /* Consecutive red links */
515  if (rbtree_node_is_red(root)) {
516  if (rbtree_node_is_red(ln) || rbtree_node_is_red(rn)) {
517  pr_err("Red violation");
518  return 0;
519  }
520  }
521 
522  lh = rbtree_tree_test(tree, ln);
523  rh = rbtree_tree_test(tree, rn);
524 
525  /* Invalid binary search tree */
526  if ((ln != NULL && tree->cmp(tree, ln, root) >= 0) || (rn != NULL && tree->cmp(tree, rn, root) <= 0)) {
527  pr_err("Binary tree violation");
528  return 0;
529  }
530 
531  /* Black height mismatch */
532  if (lh != 0 && rh != 0 && lh != rh) {
533  pr_err("Black violation");
534  return 0;
535  }
536 
537  /* Only count black links */
538  if (lh != 0 && rh != 0)
539  return rbtree_node_is_red(root) ? lh : lh + 1;
540  else
541  return 0;
