54 enum zone_type {
55  /*
56  * ZONE_DMA is used when there are devices that are not able
57  * to do DMA to all of addressable memory (ZONE_NORMAL). Then we
58  * carve out the portion of memory that is needed for these devices.
59  * The range is arch specific.
60  *
61  * Some examples
62  *
63  * Architecture Limit
64  * ---------------------------
65  * parisc, ia64, sparc <4G
66  * s390 <2G
67  * arm Various
68  * alpha Unlimited or 0-16MB.
69  *
70  * i386, x86_64 and multiple other arches
71  * <16M.
72  */
73 
74  // /// Direct memory access.
75  //ZONE_DMA,
76 
77  /*
78  * Normal addressable memory is in ZONE_NORMAL. DMA operations can be
79  * performed on pages in ZONE_NORMAL if the DMA devices support
80  * transfers to all addressable memory.
81  */
83  ZONE_NORMAL,
84 
85  /*
86  * A memory area that is only addressable by the kernel through
87  * mapping portions into its own address space. This is for example
88  * used by i386 to allow the kernel to address the memory beyond
89  * 900MB. The kernel will set up special mappings (page
90  * table entries on i386) for each page that the kernel needs to
91  * access.
92  */
94  ZONE_HIGHMEM,
