34 static inline int __push_args_on_stack(uintptr_t *esp, char *args[], char ***argsptr)
35 {
36  int argc = 0;
37  char *args_ptr[256];
38  // Count the number of arguments.
39  while (args[argc] != NULL) {
40  ++argc;
41  }
42 
43  // Prepare args with space for the terminating NULL.
44  for (int i = argc - 1; i >= 0; --i) {
45  for (int j = strlen(args[i]); j >= 0; --j) {
46  PUSH_ARG((*esp), char, args[i][j]);
47  }
48  args_ptr[i] = (char *)(*esp);
49  }
50  // Push terminating NULL.
51  PUSH_ARG((*esp), char *, (char *)NULL);
52  // Push array of pointers to the arguments.
53  for (int i = argc - 1; i >= 0; --i) {
54  PUSH_ARG((*esp), char *, args_ptr[i]);
55  }
56  (*argsptr) = (char **)(*esp);

358 pid_t sys_fork(pt_regs *f)
359 {
360  task_struct *current = scheduler_get_current_process();
361  if (current == NULL)
362  kernel_panic("There is no current process!");
363 
364  pr_debug("Forking '%s' (pid: %d)...\n", current->name, current->pid);
365 
366  // Update current process registers, they should be equal
367  // to the ones of the child process, except for eax.
368  scheduler_store_context(f, current);
369  // Allocate the memory for the process.
370  task_struct *proc = __alloc_task(current, current, current->name);
371  // Copy the father's stack, memory, heap etc... to the child process
372  proc->mm = clone_process_image(current->mm);
373  // Set the eax as 0, to indicate the child process
374  proc->thread.regs.eax = 0;
375  // Enable the interrupts.
376  proc->thread.regs.eflags = proc->thread.regs.eflags | EFLAG_IF;
377 
378  // Copy session and group id of the parent into the child
379  proc->sid = current->sid;
380  proc->gid = current->gid;
381 
382  // Active the new process.
383  scheduler_enqueue_task(proc);
384 
385  pr_debug("Forked '%s' (pid: %d, gid: %d, sid: %d)...\n", proc->name, proc->pid, proc->gid, proc->sid);
386 
387  // Return PID of child process to parent.
388  return proc->pid;
389 }

404  if (filename == NULL) {
405  pr_err("Received NULL filename.\n");
406  return -1;
407  }
408  // Get the arguments
409  origin_argv = (char **)f->ecx;
410  // Get the environment.
411  origin_envp = (char **)f->edx;
412  // Check the argument, the environment, and that at least the name is provided.
413  if (origin_argv == NULL) {
414  pr_err("sys_execve failed: must provide argv.\n");
415  return -1;
416  }
417  if (origin_argv[0] == NULL) {
418  pr_err("sys_execve failed: must provide the name.\n");
419  return -1;
420  }
421  if (origin_envp == NULL) {
422  pr_err("sys_execve failed: must provide the environment.\n");
423  return -1;
424  }
425 
426  // Save the name of the process.
427  strcpy(name_buffer, origin_argv[0]);
428 
429  // == COPY PROGRAM ARGUMENTS ==============================================
430  // Copy argv and envp to kernel memory, because all the old process memory will be discarded.
431  int argv_bytes = __count_args_bytes(origin_argv);
432  int envp_bytes = __count_args_bytes(origin_envp);
433  if ((argv_bytes < 0) || (envp_bytes < 0)) {
434  pr_err("Failed to count required memory to store arguments and environment (%d + %d).\n",
435  argv_bytes, envp_bytes);
436  return -1;
437  }

439  if (!args_mem) {

442  return -1;
443  }
444  // Copy the arguments.
445  uint32_t args_mem_ptr = (uint32_t)args_mem + (argv_bytes + envp_bytes);
446  __push_args_on_stack(&args_mem_ptr, origin_argv, &saved_argv);
447  __push_args_on_stack(&args_mem_ptr, origin_envp, &saved_envp);
448  // Check the memory pointer.
449  assert(args_mem_ptr == (uint32_t)args_mem);
450  // ------------------------------------------------------------------------
451 
452  // == INITIALIZE TASK MEMORY ==============================================
453  if (!__load_executable(filename, current, &current->thread.regs.eip)) {
454  pr_err("Failed to load executable!\n");

456  kfree(args_mem);
457  return -1;
458  }
459  // ------------------------------------------------------------------------

462  // Save the current page directory.

464 
465  // Change the page directory to point to the newly created process
466  paging_switch_directory_va(current->mm->pgd);
467
