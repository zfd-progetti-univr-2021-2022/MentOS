35 multiboot_memory_map_t *mmap_next_entry(multiboot_info_t *info,
36  multiboot_memory_map_t *entry)
37 {
38  uintptr_t next = ((uintptr_t)entry) + entry->size + sizeof(entry->size);
39  if (next >= (info->mmap_addr + info->mmap_length))
40  return NULL;
41  return (multiboot_memory_map_t *)next;
42 }

44 multiboot_memory_map_t *mmap_next_entry_of_type(multiboot_info_t *info,
45  multiboot_memory_map_t *entry,
46  uint32_t type)
47 {
48  do {
49  entry = mmap_next_entry(info, entry);

51  return entry;

72 multiboot_module_t *next_module(multiboot_info_t *info,
73  multiboot_module_t *mod)
74 {
75  multiboot_module_t *first = (multiboot_module_t *)((uintptr_t)info->mods_addr);
76  ++mod;
77  if ((mod - first) >= info->mods_count)
78  return NULL;
79  return mod;

82 void dump_multiboot(multiboot_info_t *mbi)
83 {
84  pr_debug("\n--------------------------------------------------\n");
85  pr_debug("MULTIBOOT header at 0x%x:\n", mbi);
86 
87  // Print out the flags.
88  pr_debug("%-16s = 0x%x\n", "flags", mbi->flags);
89 
90  // Are mem_* valid?
91  if (bitmask_check(mbi->flags, MULTIBOOT_FLAG_MEM)) {
92  pr_debug("%-16s = %u Kb (%u Mb)\n", "mem_lower", mbi->mem_lower,
93  mbi->mem_lower / K);
94  pr_debug("%-16s = %u Kb (%u Mb)\n", "mem_upper", mbi->mem_upper,
95  mbi->mem_upper / K);
96  pr_debug("%-16s = %u Kb (%u Mb)\n", "total",
97  mbi->mem_lower + mbi->mem_upper,
98  (mbi->mem_lower + mbi->mem_upper) / K);
99  }
100 
101  // Is boot_device valid?
102  if (bitmask_check(mbi->flags, MULTIBOOT_FLAG_DEVICE)) {
103  pr_debug("%-16s = 0x%x (0x%x)", "boot_device", mbi->boot_device);
104  switch ((mbi->boot_device) & 0xFF000000) {
105  case 0x00000000:
106  pr_debug("(floppy)\n");
107  break;
108  case 0x80000000:
109  pr_debug("(disk)\n");
110  break;
111  default:
112  pr_debug("(unknown)\n");
113  }
114  }
115 
116  // Is the command line passed?
117  if (bitmask_check(mbi->flags, MULTIBOOT_FLAG_CMDLINE)) {
118  pr_debug("%-16s = %s\n", "cmdline", (char *)mbi->cmdline);
119  }
120 
121  // Are mods_* valid?
122  if (bitmask_check(mbi->flags, MULTIBOOT_FLAG_MODS)) {
123  pr_debug("%-16s = %d\n", "mods_count", mbi->mods_count);
124  pr_debug("%-16s = 0x%x\n", "mods_addr", mbi->mods_addr);
125  multiboot_module_t *mod = first_module(mbi);
126  for (int i = 0; mod; ++i, mod = next_module(mbi, mod)) {
127  pr_debug(" [%2d] "
128  "mod_start = 0x%x, "
129  "mod_end = 0x%x, "
130  "cmdline = %s\n",
131  i, mod->mod_start, mod->mod_end, (char *)mod->cmdline);
132  }
133  }
134  // Bits 4 and 5 are mutually exclusive!
135  if (bitmask_check(mbi->flags, MULTIBOOT_FLAG_AOUT) &&
136  bitmask_check(mbi->flags, MULTIBOOT_FLAG_ELF)) {
137  kernel_panic("Both bits 4 and 5 are set.\n");
138  return;
139  }
140 
141  // Is the symbol table of a.out valid?
142  if (bitmask_check(mbi->flags, MULTIBOOT_FLAG_AOUT)) {
143  multiboot_aout_symbol_table_t *multiboot_aout_sym = &(mbi->u.aout_sym);
144  pr_debug("multiboot_aout_symbol_table: tabsize = 0x%0x, "
145  "strsize = 0x%x, addr = 0x%x\n",
146  multiboot_aout_sym->tabsize, multiboot_aout_sym->strsize,
147  multiboot_aout_sym->addr);
148  }
149 
150  // Is the section header table of ELF valid?
151  if (bitmask_check(mbi->flags, MULTIBOOT_FLAG_ELF)) {
152  multiboot_elf_section_header_table_t *multiboot_elf_sec =
153  &(mbi->u.elf_sec);
154  pr_debug("multiboot_elf_sec: num = %u, size = 0x%x,"
155  " addr = 0x%x, shndx = 0x%x\n",
156  multiboot_elf_sec->num, multiboot_elf_sec->size,
157  multiboot_elf_sec->addr, multiboot_elf_sec->shndx);
158  }
159 
160  // Are mmap_* valid?
161  if (bitmask_check(mbi->flags, MULTIBOOT_FLAG_MMAP)) {
162  pr_debug("%-16s = 0x%x\n", "mmap_addr", mbi->mmap_addr);
163  pr_debug("%-16s = 0x%x (%d entries)\n", "mmap_length",
164  mbi->mmap_length,
165  mbi->mmap_length / sizeof(multiboot_memory_map_t));
166  multiboot_memory_map_t *mmap = mmap_first_entry(mbi);
167  for (int i = 0; mmap; ++i, mmap = mmap_next_entry(mbi, mmap)) {
168  pr_debug(" [%2d] "
169  "base_addr = 0x%09x%09x, "
170  "length = 0x%09x%09x, "
171  "type = 0x%x (%s)\n",
172  i, mmap->base_addr_high, mmap->base_addr_low,
173  mmap->length_high, mmap->length_low, mmap->type,
174  mmap_type_name(mmap));
175  }
176  }
177 
178  if (bitmask_check(mbi->flags, MULTIBOOT_FLAG_DRIVE_INFO)) {
179  pr_debug("Drives: 0x%x\n", mbi->drives_length);
180  pr_debug("Addr : 0x%x\n", mbi->drives_addr);
181  }
182 
183  if (bitmask_check(mbi->flags, MULTIBOOT_FLAG_CONFIG_TABLE)) {
184  pr_debug("Config: 0x%x\n", mbi->config_table);
185  }
186 
187  if (bitmask_check(mbi->flags, MULTIBOOT_FLAG_BOOT_LOADER_NAME)) {
188  pr_debug("boot_loader_name: %s\n", (char *)mbi->boot_loader_name);
189  }
190 
191  if (bitmask_check(mbi->flags, MULTIBOOT_FLAG_APM_TABLE)) {
192  pr_debug("APM : 0x%x\n", mbi->apm_table);
193  }
194 
195  if (bitmask_check(mbi->flags, MULTIBOOT_FLAG_VBE_INFO)) {
196  pr_debug("VBE Co: 0x%x\n", mbi->vbe_control_info);
197  pr_debug("VBE Mo: 0x%x\n", mbi->vbe_mode_info);
198  pr_debug("VBE In: 0x%x\n", mbi->vbe_mode);
199  pr_debug("VBE se: 0x%x\n", mbi->vbe_interface_seg);
200  pr_debug("VBE of: 0x%x\n", mbi->vbe_interface_off);
201  pr_debug("VBE le: 0x%x\n", mbi->vbe_interface_len);
202  }
203  pr_debug("--------------------------------------------------\n");
204  pr_debug("\n");
205 }
