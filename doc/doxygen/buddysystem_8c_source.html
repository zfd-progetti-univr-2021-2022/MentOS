273 void buddy_system_init(bb_instance_t *instance,
274  const char *name,
275  void *pages_start,
276  uint32_t bbpage_offset,
277  uint32_t pages_stride,
278  uint32_t pages_count)
279 {
280  // Compute the base bb page of the struct
281  instance->base_page = ((bb_page_t *)(((uint32_t)pages_start) + bbpage_offset));
282 
283  // Save all needed page info
284  instance->bbpg_offset = bbpage_offset;
285  instance->pgs_size = pages_stride;
286  instance->size = pages_count;
287  instance->name = name;
288 
289  // Initialize all pages
290  for (uint32_t i = 0; i < pages_count; i++) {
291  bb_page_t *page = __get_page_at_index(instance, i);
292  page->flags = 0;
293  // Mark page as free
294  __bb_set_flag(page, FREE_PAGE);
295  // Initialize siblings list
296  list_head_init(&(page->location.siblings));
297  }
298 
299  // Initialize the free_lists of each area of the zone.
300  for (unsigned int order = 0; order < MAX_BUDDYSYSTEM_GFP_ORDER; order++) {
301  bb_free_area_t *area = instance->free_area + order;
302  area->nr_free = 0;
303  list_head_init(&area->free_list);
304  }
305 
306  // Current base page descriptor of the zone.
307  bb_page_t *page = instance->base_page;
308  // Address of the last page descriptor of the zone.
309  bb_page_t *last_page = __get_page_from_base(instance, page, instance->size);
310 
311  // Get the free area collecting the larges block of page frames.
312  const unsigned int order = MAX_BUDDYSYSTEM_GFP_ORDER - 1;
313  bb_free_area_t *area = instance->free_area + order;
314 
315  // Add all zone's pages to the largest free area block.
316  uint32_t block_size = 1UL << order;
317  while ((page + block_size) <= last_page) {
318  // Save the order of the page.
319  page->order = order;
320  // Set the page as root
321  __bb_set_flag(page, ROOT_PAGE);
322 
323  // Insert page as first element in the list.
324  list_head_add_tail(&page->location.siblings, &area->free_list);
325  // Increase the number of free block of the free_area_t.
326  area->nr_free++;
327 
328  page = __get_page_from_base(instance, page, block_size);
329  }
330 
331  assert(page == last_page &&
332  "Memory size is not aligned to MAX_ORDER size!");
333 }
