450 static inline int procfs_getdents(vfs_file_t *file, dirent_t *dirp, off_t doff, size_t count)
451 {
452  if (!file || !dirp)
453  return -1;
454  procfs_file_t *direntry = procfs_find_entry_inode(file->ino);
455  if (direntry == NULL) {
456  return -ENOENT;
457  }
458  if ((direntry->flags & DT_DIR) == 0)
459  return -ENOTDIR;
460  memset(dirp, 0, count);
461  int len = strlen(direntry->name);
462  size_t written = 0;
463  off_t current = 0;
464  char *parent = NULL;
465  for (off_t it = 0; (it < PROCFS_MAX_FILES) && (written < count); ++it) {
466  procfs_file_t *entry = &fs_specs.headers[it];
467  if (entry->inode == -1) {
468  continue;
469  }
470  // If the entry is the directory itself, skip.
471  if (strcmp(direntry->name, entry->name) == 0) {
472  continue;
473  }
474  // Get the parent directory.
475  parent = dirname(entry->name);
476  // Check if the entry is inside the directory.
477  if (strcmp(direntry->name, parent) != 0) {
478  continue;
479  }
480  // Skip if already provided.
481  if (current++ < doff) {
482  continue;
483  }
484  if (*(entry->name + len) == '/')
485  ++len;
486  // Write on current dirp.
487  dirp->d_ino = it;
488  dirp->d_type = entry->flags;
489  strcpy(dirp->d_name, entry->name + len);
490  dirp->d_off = sizeof(dirent_t);
491  dirp->d_reclen = sizeof(dirent_t);
492  // Increment the written counter.
493  written += sizeof(dirent_t);
494  // Move to next writing position.
495  dirp += 1;
496  }
497  return written;
