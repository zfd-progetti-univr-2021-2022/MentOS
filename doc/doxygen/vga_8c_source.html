152 static void __set_mode(vga_mode_t *vga_mode)
153 {
154  unsigned char *ptr = &vga_mode->misc;
155 
156  // Write SEQUENCER regs.
157  outportb(MISC_WRITE, *ptr);
158  ++ptr;
159 
160  // Write SEQUENCER regs.
161  for (unsigned i = 0; i < MODE_NUM_SEQ_REGS; i++) {
162  outportb(SC_INDEX, i);
163  outportb(SC_DATA, *ptr);
164  ++ptr;
165  }
166 
167  // Wnlock CRTC registers.
168  outportb(CRTC_INDEX, 0x03);
169  outportb(CRTC_DATA, inportb(CRTC_DATA) | 0x80);
170  outportb(CRTC_INDEX, 0x11);
171  outportb(CRTC_DATA, inportb(CRTC_DATA) & ~0x80);
172 
173  // Make sure they remain unlocked.
174  ptr[0x03] |= 0x80;
175  ptr[0x11] &= ~0x80;
176 
177  // write CRTC regs.
178  for (unsigned i = 0; i < MODE_NUM_CRTC_REGS; i++) {
179  outportb(CRTC_INDEX, i);
180  outportb(CRTC_DATA, *ptr);
181  ++ptr;
182  }
183 
184  // write GRAPHICS CONTROLLER regs.
185  for (unsigned i = 0; i < MODE_NUM_GC_REGS; i++) {
186  outportb(GC_INDEX, i);
187  outportb(GC_DATA, *ptr);
188  ++ptr;
189  }
190 
191  // write ATTRIBUTE CONTROLLER regs.
192  for (unsigned i = 0; i < MODE_NUM_AC_REGS; i++) {
193  inportb(INPUT_STATUS_READ);
194  outportb(AC_INDEX, i);
195  outportb(AC_WRITE, *ptr);
196  ++ptr;
197  }
198 
199  // lock 16-color palette and unblank display.
200  (void)inportb(INPUT_STATUS_READ);
201  outportb(AC_INDEX, 0x20);
202 }

246 static void __write_font(unsigned char *buf, unsigned font_height)
247 {
248  unsigned char seq2, seq4, gc4, gc5, gc6;
249  unsigned i;
250 
251  /* save registers
252 __set_plane() modifies GC 4 and SEQ 2, so save them as well */
253  outportb(SC_INDEX, 2);
254  seq2 = inportb(SC_DATA);
255 
256  outportb(SC_INDEX, 4);
257  seq4 = inportb(SC_DATA);
258  /* turn off even-odd addressing (set flat addressing)
259 assume: chain-4 addressing already off */
260  outportb(SC_DATA, seq4 | 0x04);
261 
262  outportb(GC_INDEX, 4);
263  gc4 = inportb(GC_DATA);
264 
265  outportb(GC_INDEX, 5);
266  gc5 = inportb(GC_DATA);
267  /* turn off even-odd addressing */
268  outportb(GC_DATA, gc5 & ~0x10);
269 
270  outportb(GC_INDEX, 6);
271  gc6 = inportb(GC_DATA);
272  /* turn off even-odd addressing */
273  outportb(GC_DATA, gc6 & ~0x02);
274  /* write font to plane P4 */
275  __set_plane(2);
276  /* write font 0 */
277  for (i = 0; i < 256; i++) {
278  memcpy((char *)buf, (char *)(i * 32 + 0xA0000), font_height);
279  buf += font_height;
280  }
281  /* restore registers */
282  outportb(SC_INDEX, 2);
283  outportb(SC_DATA, seq2);
284  outportb(SC_INDEX, 4);
285  outportb(SC_DATA, seq4);
286  outportb(GC_INDEX, 4);
287  outportb(GC_DATA, gc4);
288  outportb(GC_INDEX, 5);
289  outportb(GC_DATA, gc5);
290  outportb(GC_INDEX, 6);
291  outportb(GC_DATA, gc6);
292 }

323 static void __write_pixel_4(unsigned int x, unsigned int y, unsigned char color)
324 {
325  int rotation = 0;
326  int16_t t;
327  switch (rotation) {
328  case 1:
329  t = x;
330  x = driver->width - 1 - y;
331  y = t;
332  break;
333  case 2:
334  x = driver->width - 1 - x;
335  y = driver->height - 1 - y;
336  break;
337  case 3:
338  t = x;
339  x = y;
340  y = driver->height - 1 - t;
341  break;
342  }
343 
344  unsigned wd_in_bytes, off, mask, p, pmask;
345 
346  wd_in_bytes = driver->width / 8;
347  off = wd_in_bytes * y + x / 8;
348  x = (x & 7) * 1;
349  mask = 0x80 >> x;
350  pmask = 1;
351  for (p = 0; p < 4; p++) {
352  __set_plane(p);
353  if (pmask & color)
354  __write_byte(off, __read_byte(off) | mask);
355  else
356  __write_byte(off, __read_byte(off) & ~mask);
357  pmask <<= 1;
358  }
359 }

445 void vga_draw_line(unsigned int x0, unsigned int y0, unsigned int x1, unsigned int y1, unsigned char color)
446 {
447  bool_t steep = abs(y1 - y0) > abs(x1 - x0);
448  int tmp = 0;
449  if (steep) {
450  tmp = x0;
451  x0 = y0;
452  y0 = tmp;
453 
454  tmp = x1;
455  x1 = y1;
456  y1 = tmp;
457  }
458  if (x0 > x1) {
459  tmp = x0;
460  x0 = x1;
461  x1 = tmp;
462 
463  tmp = y0;
464  y0 = y1;
465  y1 = tmp;
466  }
467  int deltax = x1 - x0;
468  int deltay = abs(y1 - y0);
469  int error = deltax / 2;
470  int ystep;
471  int y = y0;
472  if (y0 < y1)
473  ystep = 1;
474  else
475  ystep = -1;
476  for (int x = x0; x < x1; x++) {
477  if (steep)
478  driver->ops->write_pixel(y, x, color);
479  else
480  driver->ops->write_pixel(x, y, color);
481  error = error - deltay;
482  if (error < 0) {
483  y = y + ystep;
484  error = error + deltax;
485  }
486  }
487 }
