24 static inline char get_task_state_char(int state)
25 {
26  if (state == 0x00) // TASK_RUNNING
27  return 'R';
28  if (state == (1 << 0)) // TASK_INTERRUPTIBLE
29  return 'S';
30  if (state == (1 << 1)) // TASK_UNINTERRUPTIBLE
31  return 'D';
32  if (state == (1 << 2)) // TASK_STOPPED
33  return 'T';
34  if (state == (1 << 3)) // TASK_TRACED
35  return 't';
36  if (state == (1 << 4)) // EXIT_ZOMBIE
37  return 'Z';
38  if (state == (1 << 5)) // EXIT_DEAD
39  return 'X';
40  return '?';
41 }

47 static ssize_t procr_read(vfs_file_t *file, char *buf, off_t offset, size_t nbyte)
48 {
49  if (file == NULL)
50  return -EFAULT;
51  // Get the entry.
52  proc_dir_entry_t *entry = (proc_dir_entry_t *)file->device;
53  if (entry == NULL)
54  return -EFAULT;
55  // Get the task.
56  task_struct *task = (task_struct *)entry->data;
57  if (task == NULL)
58  return -EFAULT;
59  // Prepare a buffer.
60  char buffer[BUFSIZ];
61  memset(buffer, 0, BUFSIZ);
62  // Call the specific function.
63  int ret = 0;
64  if (strcmp(entry->name, "cmdline") == 0)
65  ret = procr_do_cmdline(buffer, BUFSIZ, task);
66  else if (strcmp(entry->name, "stat") == 0)
67  ret = procr_do_stat(buffer, BUFSIZ, task);
68  // Perform read.
69  ssize_t it = 0;
70  if (ret == 1) {
71  size_t name_len = strlen(buffer);
72  size_t read_pos = offset;
73  if (read_pos < name_len) {
74  while ((it < nbyte) && (read_pos < name_len)) {
75  *buf++ = buffer[read_pos];
76  ++read_pos;
77  ++it;
78  }
79  }
80  }
81  return it;
82 }

104 int proc_create_entry_pid(task_struct *entry)
105 {
106  char path[PATH_MAX];
107  proc_dir_entry_t *proc_dir = NULL, *proc_entry = NULL;
108  {
109  // Create `/proc/[PID]`.
110  sprintf(path, "%d", entry->pid);
111  // Create the proc entry root directory.
112  if ((proc_dir = proc_mkdir(path, NULL)) == NULL) {
113  pr_err("[task: %d] Cannot create proc root directory `%s`.\n", entry->pid, path);
114  return -ENOENT;
115  }
116  proc_dir->data = entry;
117  }
118  {
119  // Create `/proc/[PID]/cmdline`.
120  if ((proc_entry = proc_create_entry("cmdline", proc_dir)) == NULL) {
121  pr_err("[task: %d] Cannot create proc entry `%s`.\n", entry->pid, path);
122  return -ENOENT;
123  }
124  proc_entry->sys_operations = &procv_sys_operations;
125  proc_entry->fs_operations = &procv_fs_operations;
126  proc_entry->data = entry;
127  }
128  {
129  // Create `/proc/[PID]/stat`.
130  if ((proc_entry = proc_create_entry("stat", proc_dir)) == NULL) {
131  pr_err("[task: %d] Cannot create proc entry `%s`.\n", entry->pid, path);
132  return -ENOENT;
133  }
134  proc_entry->sys_operations = &procv_sys_operations;
135  proc_entry->fs_operations = &procv_fs_operations;
136  proc_entry->data = entry;
137  }
138  return 0;
139 }

141 int proc_destroy_entry_pid(task_struct *entry)
142 {
143  // Turn the pid into string. The maximum pid is 32768, thus entry pid is at most 6 chars.
144  char pid_str[6];
145  sprintf(pid_str, "%d", entry->pid);
146  // Get the root directory.
147  proc_dir_entry_t *proc_dir = proc_dir_entry_get(pid_str, NULL);
148  if (proc_dir == NULL) {
149  pr_err("[task: %d] Cannot find proc root directory `%s`.\n", entry->pid, pid_str);
150  return -ENOENT;
151  }
152  // Destroy `/proc/[PID]/cmdline`.
153  if (proc_destroy_entry("cmdline", proc_dir)) {
154  pr_err("[task: %d] Cannot destroy proc cmdline.\n", entry->pid);
155  return -ENOENT;
156  }
157  // Destroy `/proc/[PID]/stat`.
158  if (proc_destroy_entry("stat", proc_dir)) {
159  pr_err("[task: %d] Cannot destroy proc stat.\n", entry->pid);
160  return -ENOENT;
161  }
162  // Destroy `/proc/[PID]`.
163  if (proc_rmdir(pid_str, NULL)) {
164  pr_err("[task: %d] Cannot remove proc root directory `%s`.\n", entry->pid, pid_str);
165  return -ENOENT;
166  }
167  return 0;
168 }
