119  __setup_pages(0, 0, lowmem_last_phy_page);

129 static void __get_kernel_low_high(elf_header_t *elf_hdr,
130  uint32_t *virt_low,
131  uint32_t *virt_high)
132 {
133  // Prepare a pointer to a program header.
134  elf_program_header_t *program_header;
135  // Compute the offset for accessing the program headers.
136  uint32_t offset = (uint32_t)elf_hdr + (uint32_t)elf_hdr->phoff;
137  // In this two variables we will store the start and end addresses of the segment.
138  uint32_t segment_start, segment_end;
139  // Iterate for each program header.
140  for (int i = 0; i < elf_hdr->phnum; i++) {
141  program_header = (elf_program_header_t *)(offset + elf_hdr->phentsize * i);
142  if (program_header->type == PT_LOAD) {
143  // Take the start and end addresses of the segment from the program header.
144  segment_start = program_header->vaddr;
145  segment_end = segment_start + program_header->memsz;
146  // Take the lowest and highest virtual address.
147  *virt_low = min(*virt_low, segment_start);
148  *virt_high = max(*virt_high, segment_end);
149  }
150  }
151 }

156 static inline uint32_t __get_address_after_modules(multiboot_info_t *header)
157 {
158  // We set by default the address to the ending physical address
159  // of the bootloader.
160  uint32_t addr = boot_info.bootloader_phy_end;
161  // Get the pointer to the mods.
162  multiboot_module_t *mod = (multiboot_module_t *)header->mods_addr;
163  for (int i = 0; (i < header->mods_count) && (i < MAX_MODULES); ++i, ++mod) {
164  addr = max(max(addr, mod->mod_start), mod->mod_end);
165  }
166  return addr;
167 }

171 static inline void __relocate_kernel_image(elf_header_t *elf_hdr)
172 {
173  // Support variables.
174  elf_program_header_t *program_header;
175  char *kernel_start, *virtual_address, *physical_address;
176  uint32_t offset, valid_size;
177 
178  // Get the elf file starting address.
179  kernel_start = (char *)elf_hdr;
180  // Compute the offset for accessing the program headers.
181  offset = (uint32_t)kernel_start + (uint32_t)elf_hdr->phoff;
182  // Iterate over the program headers.
183  for (int i = 0; i < elf_hdr->phnum; i++) {
184  // Get the program header.
185  program_header = (elf_program_header_t *)(offset + elf_hdr->phentsize * i);
186  // Get the virtual address of the program header.
187  virtual_address = (char *)program_header->vaddr;
188  // Get the physical address of the program header.
189  physical_address = (char *)(kernel_start + program_header->offset);
190  // Move only the loadable segments.
191  if (program_header->type == PT_LOAD) {
192  // Get the valid size of the segment by taking the minimum between
193  // the size in bytes of the segment in the file image, in memory.
194  valid_size = min(program_header->filesz, program_header->memsz);
195  // Copy the physical data of the image to the corresponding virtual address.
196  for (int j = 0; j < valid_size; j++)
197  virtual_address[j] = physical_address[j];
198  // Set to 0 parts not present in memory!
199  for (int j = valid_size; j < program_header->memsz; j++)
200  virtual_address[j] = 0;
201  }
202  }
203 }

209 void boot_main(uint32_t magic, multiboot_info_t *header, uint32_t esp)
210 {
211  __debug_puts("\nbootloader: Start...\n");
212  elf_header_t *elf_hdr = (elf_header_t *)LDVAR(kernel_bin);
213 
214  // Get the physical addresses of where the kernel starts and ends.
215  uint32_t boot_start = (uint32_t)_bootloader_start;
216  uint32_t boot_end = (uint32_t)_bootloader_end;
217 
218  // Extract the lowest and highest address of the kernel.
219  uint32_t kernel_virt_low = 0xFFFFFFFF;
220  uint32_t kernel_virt_high = 0;
221  __get_kernel_low_high(elf_hdr, &kernel_virt_low, &kernel_virt_high);
222 
223  // Initialize the boot_info_t structure.
224  __debug_puts("bootloader: Initializing the boot_info structure...\n");
225  boot_info.magic = magic;
226  boot_info.bootloader_phy_start = boot_start;
227  boot_info.bootloader_phy_end = boot_end;
228  boot_info.kernel_start = kernel_virt_low;
229  boot_info.kernel_end = kernel_virt_high;
230  boot_info.kernel_size = kernel_virt_high - kernel_virt_low;
231  boot_info.multiboot_header = header;
232 
233  // Get the address after the modules.
234  boot_info.module_end = __get_address_after_modules(header);
235 
236  // Get the starting address of the physical pages at the end of the modules.
237  uint32_t kernel_phy_page_start = __align_rup(boot_info.module_end, PAGE_SIZE);
238  // Get the starting address of the virtual pages.
239  uint32_t kernel_virt_page_start = __align_rdown(kernel_virt_low, PAGE_SIZE);
240 
241  // Compute the absolute offset of the first virtual page, by subtracting
242  // the starting address of the virtual pages and the lowest virtual address
243  // of the kernel.
244  uint32_t kernel_page_offset = kernel_virt_page_start - kernel_virt_low;
245 
246  // If we add the offset we computed earlier to the physical address where
247  // the modules ends, we obtain the starting address of the physical memory.
248  boot_info.kernel_phy_start = kernel_phy_page_start + kernel_page_offset;
249  // The ending address of the physical memory is just the start plus the
250  // size of the kernel (virt_high - virt_low).
251  boot_info.kernel_phy_end = boot_info.kernel_phy_start + boot_info.kernel_size;
252 
253  boot_info.lowmem_phy_start = __align_rup(boot_info.kernel_phy_end, PAGE_SIZE);
254  boot_info.lowmem_phy_end = 896 * 1024 * 1024; // 896 MB of low memory max
255 
256  uint32_t lowmem_size = boot_info.lowmem_phy_end - boot_info.lowmem_phy_start;
257 
258  boot_info.lowmem_start = __align_rup(boot_info.kernel_end, PAGE_SIZE);
259  boot_info.lowmem_end = boot_info.lowmem_start + lowmem_size;
260 
261  boot_info.highmem_phy_start = boot_info.lowmem_phy_end;
262  boot_info.highmem_phy_end = header->mem_upper * 1024;
263  boot_info.stack_end = boot_info.lowmem_end;
264 
265  // Setup the page directory and page tables for the boot.
266  __debug_puts("bootloader: Setting up paging...\n");
267  __setup_boot_paging();
268 
269  // Switch to the newly created page directory.
270  __debug_puts("bootloader: Switching page directory...\n");
271  paging_switch_directory(&boot_pgdir);
272 
273  // Enable paging.
274  __debug_puts("bootloader: Enabling paging...\n");
275  paging_enable();
276 
277  // Reserve space for the kernel stack at the end of lowmem.
278  boot_info.stack_base = boot_info.lowmem_end;
279  boot_info.lowmem_phy_end = boot_info.lowmem_phy_end - KERNEL_STACK_SIZE;
280  boot_info.lowmem_end = boot_info.lowmem_end - KERNEL_STACK_SIZE;
281 
282  __debug_puts("bootloader: Relocating kernel image...\n");
283  __relocate_kernel_image(elf_hdr);
284 
285 #if 0
286  for (int i = 0; i < elf_hdr->shnum; i++) {
287  struct elf_section_header *section_header =
288  (elf_section_header_t *)(LDVAR(kernel_bin) + elf_hdr->shoff + elf_hdr->shentsize * i);
289  for (int j = 0; j < section_header->; j++) {
290  ((char *)section_header->vaddr)[j] = (LDVAR(kernel_bin) + section_header->offset)[j];
291  }
292  }
293 #endif
294 
295  __debug_puts("bootloader: Calling `boot_kernel`...\n\n");
296  boot_kernel(boot_info.stack_base, elf_hdr->entry, &boot_info);
297 }
