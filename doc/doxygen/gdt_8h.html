#define GDT_PRESENT   128U
 Present bit. This must be 1 for all valid selectors.
 
#define GDT_KERNEL   0U
 Sets the 2 privilege bits (ring level) to 0 = highest (kernel).
 
#define GDT_USER   96U
 Sets the 2 privilege bits (ring level) to 3 = lowest (user applications).
 
#define GDT_S   16U
 Descriptor type. This bit should be set for code or data segments and should be cleared for system segments (eg. a Task State Segment)
 
#define GDT_EX   8U
 Executable bit. If 1 code in this segment can be executed, ie. a code selector. If 0 it is a data selector.
 
#define GDT_DC   4U
 Direction bit/Conforming bit.
 
#define GDT_RW   2U
 Readable bit/Writable bit.
 
#define GDT_AC   1U
 Accessed bit. Just set to 0. The CPU sets this to 1 when the segment is accessed.
 
#define GDT_CODE   (GDT_S | GDT_EX)
 Identifies an executable code segment.
 
#define GDT_DATA   (GDT_S | GDT_RW)
 Identifies a writable data segment.
 
#define GDT_GRANULARITY   128U
 Granularity bit. If 0 the limit is in 1 B blocks (byte granularity), if 1 the limit is in 4 KiB blocks (page granularity).
 
#define GDT_OPERAND_SIZE   64U
 Size bit. If 0 the selector defines 16 bit protected mode. If 1 it defines 32 bit protected mode. You can have both 16 bit and 32 bit selectors at once.
 
#define IDT_PADDING   14U
 Used in IDT for padding.

void init_gdt ( )

Initialise the GDT.

This will setup the special GDT pointer, set up the first 3 entries in our GDT, and then finally call gdt_flush() in our assembler file in order to tell the processor where the new GDT is and update the new segment registers.
