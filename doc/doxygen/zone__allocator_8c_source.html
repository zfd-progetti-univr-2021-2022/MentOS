104 {
105  // Get the corresponding zone.

107  assert(zone && "Failed to retrieve the zone given the gfp_mask!");
108  // Get the last free area list of the buddy system.

110  assert(area && "Failed to retrieve the last free_area for the given zone!");

209 static void zone_init(char *name, int zone_index, uint32_t adr_from, uint32_t adr_to)
210 {
211  assert((adr_from < adr_to) && "Inserted bad block addresses!");
212  assert(((adr_from & 0xfffff000) == adr_from) && "Inserted bad block addresses!");
213  assert(((adr_to & 0xfffff000) == adr_to) && "Inserted bad block addresses!");
214  assert((zone_index < contig_page_data->nr_zones) && "The index is above the number of zones.");
215  // Take the zone_t structure that correspondes to the zone_index.
216  zone_t *zone = contig_page_data->node_zones + zone_index;
217  assert(zone && "Failed to retrieve the zone.");
218  // Number of page frames in the zone.
219  size_t num_page_frames = (adr_to - adr_from) / PAGE_SIZE;
220  // Index of the first page frame of the zone.
221  uint32_t first_page_frame = adr_from / PAGE_SIZE;
222  // Update zone info.
223  zone->name = name;
224  zone->size = num_page_frames;
225  zone->free_pages = num_page_frames;
226  zone->zone_mem_map = mem_map + first_page_frame;
227  zone->zone_start_pfn = first_page_frame;
228  // Dump the information.
229  pr_debug("ZONE %s, first page: %p, last page: %p, npages:%d\n", zone->name,
230  zone->zone_mem_map, zone->zone_mem_map + zone->size, zone->size);
231  // Set to zero all page structures.
232  memset(zone->zone_mem_map, 0, zone->size * sizeof(page_t));
233  // Initialize the buddy system for the new zone.
234  buddy_system_init(&zone->buddy_system,
235  name,
236  zone->zone_mem_map,
237  BBSTRUCT_OFFSET(page_t, bbpage),
238  sizeof(page_t),
239  num_page_frames);
240  buddy_system_dump(&zone->buddy_system);
241 }

249 unsigned int find_nearest_order_greater(uint32_t base_addr, uint32_t amount)
250 {
251  uint32_t start_pfn = base_addr / PAGE_SIZE;
252  uint32_t end_pfn = (base_addr + amount + PAGE_SIZE - 1) / PAGE_SIZE;
253  // Get the number of pages.
254  uint32_t npages = end_pfn - start_pfn;
255  // Find the fitting order.
256  unsigned int order = 0;
257  while ((1UL << order) < npages) {
258  ++order;
259  }
260  return order;
261 }
