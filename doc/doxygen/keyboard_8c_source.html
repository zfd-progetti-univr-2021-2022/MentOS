193  if (bitmask_check(kflags, KBD_NUM_LOCK)) {
194  character = keymap->numlock[scancode];
195  }
196  if ((character <= 0) && bitmask_check(kflags, KBD_RIGHT_ALT) && (bitmask_check(kflags, KBD_LEFT_SHIFT | KBD_RIGHT_SHIFT))) {
197  if (scancode == KEY_LEFT_BRAKET) {
198  character = '{';
199  } else if (scancode == KEY_RIGHT_BRAKET) {
200  character = '}';
201  }
202  }
203  if ((character <= 0) && bitmask_check(kflags, KBD_RIGHT_ALT)) {

216  }

219  }
