253 static void __page_handle_cow(page_table_entry_t *entry)
254 {
255  // Check if the page is Copy On Write (COW).
256  if (entry->kernel_cow) {
257  // Set the entry is no longer COW.
258  entry->kernel_cow = 0;
259  // Check if the entry is not present (allocated).
260  if (!entry->present) {
261  // Allocate a new page.
262  page_t *page = _alloc_pages(GFP_HIGHUSER, 0);
263  // Clear the new page.
264  uint32_t vaddr = virt_map_physical_pages(page, 1);
265  memset((void *)vaddr, 0, PAGE_SIZE);
266  // Unmap the virtual address.
267  virt_unmap(vaddr);
268  // Set it as current table entry frame.
269  entry->frame = get_physical_address_from_page(page) >> 12U;
270  // Set it as allocated.
271  entry->present = 1;
272  return;
273  }
274  }
275  kernel_panic("Page not cow!");
276 }

313 void page_fault_handler(pt_regs *f)
314 {
315  // Here you will find the `Demand Paging` mechanism.
316  // From `Understanding The Linux Kernel 3rd Edition`:
317  // The term demand paging denotes a dynamic memory allocation
318  // technique that consists of deferring page frame allocation
319  // until the last possible momentuntil the process attempts
320  // to address a page that is not present in RAM, thus causing
321  // a Page Fault exception.
322 
323  // First, read the linear address that caused the Page Fault.
324  // When the exception occurs, the CPU control unit stores that
325  // value in the cr2 control register.
326  uint32_t faulting_addr;
327  asm volatile("mov %%cr2, %0"
328  : "=r"(faulting_addr));
329  // Get the physical address of the current page directory.
330  uint32_t phy_dir = (uint32_t)paging_get_current_directory();
331  // Get the page directory.
332  page_directory_t *lowmem_dir = (page_directory_t *)get_lowmem_address_from_page(get_page_from_physical_address(phy_dir));
333  // Get the directory entry.
334  page_dir_entry_t *direntry = &lowmem_dir->entries[faulting_addr / (1024U * PAGE_SIZE)];
335  // TODO: Panic only if page is in kernel memory, else abort process with sigsegv
336  if (!direntry->present) {
337  __page_fault_panic(f, faulting_addr);
338  }
339  // Get the physical address of the page table.
340  uint32_t phy_table = direntry->frame << 12U;
341  // Get the page table.
342  page_table_t *lowmem_table = (page_table_t *)get_lowmem_address_from_page(get_page_from_physical_address(phy_table));
343  // Get the entry inside the table that caused the fault.
344  uint32_t table_index = (faulting_addr / PAGE_SIZE) % 1024U;
345  // Get the corresponding page table entry.
346  page_table_entry_t *entry = &lowmem_table->pages[table_index];
347  // There was a page fault on a virtual mapped address,
348  // so we must first update the original mapped page
349  if (virtual_check_address(faulting_addr)) {
350  // Get the original page table entry from the virtually mapped one.
351  page_table_entry_t *orig_entry = (page_table_entry_t *)(*(uint32_t *)entry);
352  // Check if the page is Copy on Write (CoW).
353  __page_handle_cow(orig_entry);
354  // Update the page table entry frame.
355  entry->frame = orig_entry->frame;
356  // Update the entry flags.
357  __set_pg_table_flags(entry, MM_PRESENT | MM_RW | MM_GLOBAL | MM_COW | MM_UPDADDR);
358  } else {
359  // Check if the page is Copy on Write (CoW).
360  __page_handle_cow(entry);
361  }
362  // Invalidate the page table entry.
363  paging_flush_tlb_single(faulting_addr);
364 }

403 static pg_iter_entry_t __pg_iter_next(page_iterator_t *iter)
404 {
405  pg_iter_entry_t result = {
406  .entry = &iter->table->pages[iter->pfn % 1024],
407  .pfn = iter->pfn
408  };
409 
410  if (++iter->pfn % 1024 == 0) {
411  // Create a new page only if we haven't reached the end

413  // so we can easily know when we've skipped the last page by checking
414  // if the address % PAGE_SIZE is equal to zero.

417  __set_pg_entry_frame(iter->entry, iter->table);
418  }
