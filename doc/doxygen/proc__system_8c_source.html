29 {

33  if (entry == NULL)
34  return -EFAULT;

38  // Call the specific function.
39  int ret = 0;

42  else if (strcmp(entry->name, "version") == 0)
43  ret = procs_do_version(buffer, BUFSIZ);

49  ret = procs_do_meminfo(buffer, BUFSIZ);
50  else if (strcmp(entry->name, "stat") == 0)
51  ret = procs_do_stat(buffer, BUFSIZ);
52  // Perform read.
53  ssize_t it = 0;
54  if (ret == 0) {
55  size_t name_len = strlen(buffer);
56  size_t read_pos = offset;
